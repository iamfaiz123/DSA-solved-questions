#include<bits.stdc++.h>
using namespace std;
void function(int* arr,int n)
{    int j =0;
    for(auto i = 0; i < n ;i++)
    {
        if(arr[i]<0)
        {
            if(i!=j) j++;
            else
            {
            swap(arr[i],arr[j]);
            j++;
            }
        }
    }
}
